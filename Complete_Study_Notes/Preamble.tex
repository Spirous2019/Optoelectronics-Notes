    basicstyle       = \ttfamily\small\color{black},

    keywordstyle     = \color{codeKeyword}\bfseries,
    % --- self / True / False / None / builtins / functions ---

    stringstyle      = \color{codeString},
    % --- Line numbers ---
    numberstyle      = \tiny\ttfamily\color{codeLineNum},
    numbers          = left,
    numbersep        = 12pt,
    % --- Complete purge of default listings Python classes ---
    % 1. Strip ALL class 2 built-in functions the package knows
    classoffset      = 1,
    % 2. Empty out the default lists for keywords 2, 3, etc.
    keywords         = [2]{},
    keywords         = [3]{},
    keywords         = [4]{},
    keywords         = [5]{},
    % 3. Force the style of these stripped categories to black anyway just in case
    keywordstyle     = [2]\color{black},
    keywordstyle     = [3]\color{black},
    keywordstyle     = [4]\color{black},
    keywordstyle     = [5]\color{black},
    % 4. Force default emph dictionary to black
    emphstyle        = \color{black},
    classoffset      = 0,
    % --- Layout ---

    tabsize          = 4,

% --- Key Result Block ---
% Technique: \maketag@@@ is temporarily redefined so the inline right-side
% equation number vanishes and the raw digits are stored in \@keyresulttag.
% That value is then rendered as a black pill badge at the top-left corner.
% The equation counter, \label, and \eqref all work normally.
\makeatletter
\newcommand{\@keyresulttag}{}%
\newtcolorbox{keyresult}{%
    enhanced, sharp corners, breakable,
    before upper = {%
        \setlength\abovedisplayskip{5pt}%
        \setlength\abovedisplayshortskip{5pt}%
        \setlength\belowdisplayskip{5pt}%
        \setlength\belowdisplayshortskip{5pt}%
        \def\Hy@raisedlink####1{\vbox to 0pt{\vskip-60pt \hbox{####1}\vss}}%
        \def\tagform@####1{\maketag@@@{####1}}%
        \def\maketag@@@####1{\gdef\@keyresulttag{####1}}%
    },
    boxrule         = 0pt,
    frame hidden,
    colback         = black!6!white,
    borderline west = {3.5pt}{0pt}{black},
    left            = 12pt,
    right           = 12pt,
    top             = 26pt,
    bottom          = 8pt,
    before skip     = 12pt,
    after skip      = 12pt,
    overlay = {%
        \node[anchor=north west,
              fill=black, text=white,
              font=\small\bfseries,
              inner xsep=7pt, inner ysep=3.5pt,
              rounded corners=3pt]
          at ([xshift=10pt, yshift=-8pt]frame.north west)
          {\@keyresulttag};
    },
}
\makeatother
